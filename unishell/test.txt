var stop false;
var input "";
var objs [];

while '(!$stop)' begin
	exp ($input = #input);

	if ($input == "add") begin
		var name "";
		var age 0;

		print %nl "name: ";
		exp ($name = #input);

		print %nl "age: ";
		exp ($age = (#input)@number);

		exp $objs@append[{["name", new $name], ["age", new $age]}];

		del name;
		del age;
	end;

	if ($input == "del") begin
		var who "";

		print %nl "who?: ";
		exp ($who = #input);

		var found false;
		var item null;

		foreach $item $objs begin
			if ($item@at["name"] == $who) begin
				print "removing: " $item;

				exp ($objs@eraseAt[$count]) 
					($found = true);
			end;
		end;

		if (!$found) begin
			print "couldn't find " clr:wgreen: $who clr:std:;
		end;
	end;

	if ($input == "view") begin
		print $objs;
	end;

	if ($input == "exit") begin
		exp ($stop = true);
	end;
end;

del stop;
del input;
del objs;
